⚙ วิธีเชื่อม Google Sheets
1สร้าง Google Sheet ใหม่

2เปิด Apps Script
ใน Google Sheet → เมนู Extensions → Apps Script
3วางโค้ดด้านล่าง (แทนที่โค้ดเดิมทั้งหมด)
คลิก Copy แล้ววางใน Editor แล้วกด Save (Ctrl+S)
function doPost(e) {
  try {
    var ss   = SpreadsheetApp.getActiveSpreadsheet();
    var sh   = ss.getSheetByName("บันทึกเวลา") || ss.insertSheet("บันทึกเวลา");

    // สร้าง header ถ้ายังไม่มี
    if (sh.getLastRow() === 0) {
      sh.appendRow([
        "วันที่บันทึก","เวลาเริ่มต้น","วันที่ผลลัพธ์",
        "เวลาผลลัพธ์","หมวดหมู่","ระยะเวลา","ประเภท"
      ]);
      sh.getRange(1,1,1,7).setFontWeight("bold")
        .setBackground("#4a4af0").setFontColor("#ffffff");
      sh.setFrozenRows(1);
    }

    var d = JSON.parse(e.postData.contents);
    sh.appendRow([
      d.savedAt, d.startTime, d.resultDate,
      d.resultTime, d.category, d.duration, d.operation
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({status:"ok"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:"error",msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:"ready"}))
    .setMimeType(ContentService.MimeType.JSON);
}
